    <script>
      let coins = 0;
      let spins = 0;
      const spinBtn = document.getElementById('spinBtn');
      const wheel = document.getElementById('wheel');
      const withdrawArea = document.getElementById('withdrawArea');
      const lockMsg = document.getElementById('lockMsg');

      function spinWheel() {
        spinBtn.disabled = true;
        // Random rotation animation
        const randomDeg = Math.floor(Math.random() * 360) + 1800;
        wheel.style.transform = `rotate(${randomDeg}deg) `;
        setTimeout(() => {
          // Update Logic
          coins += 100;
          spins += 1;
          updateDisplay();
          checkUnlockStatus();
          spinBtn.disabled = false;
          wheel.style.transition = 'none';
          wheel.style.transform = 'rotate(0deg)';
          setTimeout(() => wheel.style.transition = 'transform 3s cubic-bezier(0.1, 0, 0.1, 1)', 50);
          alert("Congratulations! You won 100 Coins.");
        }, 3500);
      }

      function updateDisplay() {
        document.getElementById('coinCount').innerText = coins;
        document.getElementById('spinCount').innerText = spins;
        document.getElementById('dollarValue').innerText = "$" + (coins / 1000).toFixed(2);
      }

      function checkUnlockStatus() {
        if (spins >= 100) {
          withdrawArea.style.display = 'block';
          lockMsg.style.display = 'none';
        }
      }

      function processWithdrawal() {
        const bName = document.getElementById('bankName').value;
        const accNum = document.getElementById('accNumber').value;
        if (coins < 1000) {
          alert("Minimum 1000 coins ($1) required!");
          return;
        }
        if (bName && accNum) {
          alert("Withdrawal Request Sent!\nBank: " + bName + "\nAmount: $" + (coins / 1000).toFixed(2) + "\nStatus: Pending (24-48h)");
          coins = 0;
          updateDisplay();
        } else {
          alert("Please fill all bank details.");
        }
      }
    </script>
